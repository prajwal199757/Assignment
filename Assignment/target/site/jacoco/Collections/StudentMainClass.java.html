package Collections;

import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;

public class StudentMainClass {
    public static void main(String[] args) {


        ArrayList<Student> students = new ArrayList<Student>();

        students.add(new Student(1, "Renu", "GHY"));
        students.add(new Student(2, "Ben", "Banglore"));
        students.add(new Student(3, "Joe", "Hyderabad"));
        students.add(new Student(4, "Stabin", "Banglore"));
        students.add(new Student(5, "Ariana", "Kolkata"));
        students.add(new Student(6, "David", "GHY"));

        /** Display all the students **/
        for (Student st : students) {
            System.out.println(st);
        }

        //students.stream().filter(s->s.getName().equals())

        //System.out.println(students.stream().filter(s->s.getName());

        /** sorted by Name using collection **/
        Collections.sort(students, (s1, s2) -> s1.getName().compareTo(s2.getName()));
        System.out.println("After sorted by Name" + students);

        /** sorted by city using collection **/

        Collections.sort(students, (t1, t2) -> t1.getCity().compareTo(t2.getCity()));
        System.out.println("After sorted by city " + students);

        /** find By id using stream **/
        //System.out.println("found by ID " + students.stream().filter(s -> s.getId() == 3).collect(Collectors.toList()).get(0));
        HashMap<ArrayList<Student>, Integer> mymap = new HashMap<>();
        mymap.put(students, 5);
        boolean st = mymap.containsKey(5);
        if (st == true) {
            System.out.println(mymap);
        } else {
            System.out.println("not found");
        }


    }
}
